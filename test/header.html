const picker = new Litepicker({ element: document.getElementById("inline-picker"), inlineMode: true, singleMode: true, autoApply: true, showTime: true, timeFormat: "HH:mm", format: "YYYY-MM-DD HH:mm", // Có cả ngày + giờ setup: (picker) => { picker.on("selected", (date) => { const year = date.getFullYear(); const month = date.getMonth() + 1; const day = date.getDate(); const hours = date.getHours().toString().padStart(2, "0"); const minutes = date.getMinutes().toString().padStart(2, "0"); // Ghi vào ô input nếu checkbox được chọn if (document.getElementById("dueCheckbox").checked) { document.getElementById( "dueDateInput" ).value = `${month}/${day}/${year}`; document.getElementById("dueTimeInput").value = `${hours}:${minutes}`; } }); }, });
